import {
    flexRender,
    getCoreRowModel,
    getPaginationRowModel,
    getFilteredRowModel,
    getSortedRowModel,
    useReactTable,
    type ColumnDef,
    type PaginationState,
} from "@tanstack/react-table";
import ArrowDownwardIcon from "@mui/icons-material/ArrowDownward";
import ArrowUpwardIcon from "@mui/icons-material/ArrowUpward";
import UnfoldMoreIcon from "@mui/icons-material/UnfoldMore";
import { Table, Button, Form, InputGroup, Row } from "react-bootstrap";
import { useEffect, useState } from "react";

export default function BasicTable<T>({
    data,
    columns,
    rightActions,
}: {
    data: T[];
    columns: ColumnDef<T, any>[];
    title?: string;
    rightActions?: React.ReactNode;
}) {
    const [globalFilter, setGlobalFilter] = useState("");

    const [pagination, setPagination] = useState<PaginationState>({
        pageIndex: 0,
        pageSize: 10,
    });

    const table = useReactTable({
        data,
        columns,
        state: {
            globalFilter,
            pagination,
        },
        onGlobalFilterChange: setGlobalFilter,
        onPaginationChange: setPagination,
        getCoreRowModel: getCoreRowModel(),
        getFilteredRowModel: getFilteredRowModel(),
        getSortedRowModel: getSortedRowModel(),
        getPaginationRowModel: getPaginationRowModel(),
        manualPagination: false,
    });

    useEffect(() => {
        const maxPage = Math.max(0, table.getPageCount() - 1);
        if (pagination.pageIndex > maxPage) {
            setPagination((p) => ({ ...p, pageIndex: maxPage }));
        }
        // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [data.length, pagination.pageSize]);

    const start =
        data.length === 0 ? 0 : pagination.pageIndex * pagination.pageSize + 1;
    const end = Math.min(
        (pagination.pageIndex + 1) * pagination.pageSize,
        data.length,
    );

    return (
        <div className="basic-listing">
            <style>{`
                .sticky-column {
                    position: sticky;
                    right: 0;
                    z-index: 10;
                    background-color: #fff !important;
                    border-left: 1px solid #dee2e6;
                }
                thead .sticky-column {
                    background-color: #EEF0F7 !important;
                }
                .sticky-action {
                    box-shadow: -4px 0 6px rgba(0, 0, 0, 0.1);
                }
                .table td,
                .table th {
                    background-clip: padding-box;
                }
            `}</style>

            {/* HEADER */}
            <div className="d-flex align-items-center justify-content-between mb-2 gap-2 flex-wrap">
                <InputGroup style={{ maxWidth: 280 }}>
                    <InputGroup.Text>🔍</InputGroup.Text>
                    <Form.Control
                        placeholder="Search ..."
                        value={globalFilter}
                        onChange={(e) => {
                            setGlobalFilter(e.target.value);
                            setPagination((p) => ({ ...p, pageIndex: 0 }));
                        }}
                    />
                </InputGroup>
                {rightActions && <div>{rightActions}</div>}
            </div>

            {/* TABLE */}
            <div className="table-responsive">
                <Table
                    className="table table-centered align-middle table-custom-effect table-nowrap mb-0"
                    hover
                >
                    <thead className="table-light">
                        {table.getHeaderGroups().map((hg) => (
                            <tr key={hg.id}>
                                {hg.headers.map((h, index) => {
                                    const isLastColumn = index === hg.headers.length - 1;
                                    const isActionColumn =
                                        h.column.columnDef.header?.toString().toLowerCase() === "action" ||
                                        h.column.columnDef.header?.toString().toLowerCase() === "actions";
                                    const shouldBeSticky = isLastColumn && isActionColumn;
                                    const isSorted = h.column.getIsSorted();
                                    const canSort = h.column.getCanSort();
                                    const sortIcon =
                                        isSorted === "asc" ? (
                                            <ArrowUpwardIcon fontSize="small" />
                                        ) : isSorted === "desc" ? (
                                            <ArrowDownwardIcon fontSize="small" />
                                        ) : (
                                            <UnfoldMoreIcon fontSize="small" />
                                        );

                                    return (
                                        <th
                                            key={h.id}
                                            className={`align-middle ${shouldBeSticky ? "sticky-column sticky-action" : ""}`}
                                        >
                                            {!h.isPlaceholder && (
                                                <div
                                                    className="d-flex align-items-center"
                                                    style={{
                                                        gap: 4,
                                                        cursor: canSort ? "pointer" : "default",
                                                    }}
                                                    onClick={
                                                        canSort ? h.column.getToggleSortingHandler() : undefined
                                                    }
                                                >
                                                    <span>
                                                        {flexRender(h.column.columnDef.header, h.getContext())}
                                                    </span>
                                                    {canSort && sortIcon}
                                                </div>
                                            )}
                                        </th>
                                    );
                                })}
                            </tr>
                        ))}
                    </thead>

                    <tbody>
                        {table.getRowModel().rows.map((row) => (
                            <tr key={row.id}>
                                {row.getVisibleCells().map((cell, index) => {
                                    const isLastColumn = index === row.getVisibleCells().length - 1;
                                    const isActionColumn =
                                        cell.column.columnDef.header?.toString().toLowerCase() === "action" ||
                                        cell.column.columnDef.header?.toString().toLowerCase() === "actions";
                                    const shouldBeSticky = isLastColumn && isActionColumn;

                                    return (
                                        <td
                                            key={cell.id}
                                            className={shouldBeSticky ? "sticky-column sticky-action" : ""}
                                            style={{
                                                maxWidth: cell.column.columnDef.maxSize ?? "none",
                                                whiteSpace: "nowrap",
                                                overflow: "hidden",
                                                textOverflow: "ellipsis",
                                            }}
                                        >
                                            {flexRender(cell.column.columnDef.cell, cell.getContext())}
                                        </td>
                                    );
                                })}
                            </tr>
                        ))}

                        {data.length === 0 && (
                            <tr>
                                <td
                                    colSpan={columns.length}
                                    className="text-center py-4 text-muted"
                                >
                                    No data found
                                </td>
                            </tr>
                        )}
                    </tbody>
                </Table>
            </div>

            {/* FOOTER */}
            <Row className="align-items-center mt-0 py-3 gy-2 mx-0">
                <div className="d-flex align-items-center justify-content-between flex-wrap gap-2">
                    <div className="text-muted" style={{ fontSize: 14 }}>
                        Showing {start}–{end} of {data.length} results
                    </div>

                    <div className="d-flex align-items-center gap-2">
                        <Form.Select
                            size="sm"
                            value={pagination.pageSize}
                            onChange={(e) =>
                                setPagination((p) => ({
                                    ...p,
                                    pageSize: Number(e.target.value),
                                    pageIndex: 0,
                                }))
                            }
                            style={{ width: "auto" }}
                        >
                            {[5, 10, 20, 50].map((n) => (
                                <option key={n} value={n}>
                                    Show {n}
                                </option>
                            ))}
                        </Form.Select>

                        <Button
                            size="sm"
                            variant="outline-secondary"
                            onClick={() => table.previousPage()}
                            disabled={!table.getCanPreviousPage()}
                        >
                            Prev
                        </Button>

                        <Button
                            size="sm"
                            variant="outline-secondary"
                            onClick={() => table.nextPage()}
                            disabled={!table.getCanNextPage()}
                        >
                            Next
                        </Button>
                    </div>
                </div>
            </Row>
        </div>
    );
}
